<body>

  <h1>Google Sheets Data Manager</h1>


  <input type="text" id="newEntry" placeholder="Enter new data" />

  <button onclick="addEntry()">Add Entry</button>


  <div id="entries">

    <h2>Entries from Google Sheet:</h2>

    <ul id="entryList"></ul>

  </div>


  <script>

    const WEB_APP_URL = 'https://script.google.com/macros/s/AKfycbzz0zfTAYg8BO4XCamVZ27jzhPFswWr70pii4WXgkFdyRwul63aiT-xz8sCPO9W88AGvw/exec'; // Replace with your actual URL


    // Function to add a new entry

    async function addEntry() {

      const data = document.getElementById('newEntry').value;

      if (!data) {

        alert('Please enter some data!');

        return;

      }

      try {

        await fetch(`${WEB_APP_URL}?action=addEntry&data=${encodeURIComponent(data)}`);

        document.getElementById('newEntry').value = '';

        fetchEntries(); // Refresh list

      } catch (error) {

        console.error('Error adding entry:', error);

      }

    }


    // Function to fetch all entries

    async function fetchEntries() {

      try {

        const response = await fetch(`${WEB_APP_URL}?action=getEntries`);

        const entries = await response.json();

        const entryList = document.getElementById('entryList');

        entryList.innerHTML = ''; // Clear existing entries

        entries.forEach(entry => {

          const listItem = document.createElement('li');

          listItem.textContent = entry[0]; // Assuming single-column entries

          entryList.appendChild(listItem);

        });

      } catch (error) {

        console.error('Error fetching entries:', error);

      }

    }


    // Fetch entries on page load

    fetchEntries();

  </script>
